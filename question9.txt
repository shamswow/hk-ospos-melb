create or replace FUNCTION L2_SF_Update(pSalary INTEGER) -- the pSalary is the parameter fed in from the anonymous block
RETURN INTEGER AS 
vCount INTEGER;
BEGIN
  vCount := 0;
  SELECT COUNT(*) -- Here i'm counting the number of rows that have a salary over the defined parameter
  INTO vCount
  FROM EMPLOYEE
  WHERE Salary > pSalary;
  UPDATE EMPLOYEE -- Here i'm setting any salary over the parameter to zero
  SET Salary = 0 
  WHERE Salary > pSalary;
  RETURN vCount; -- here i'm returning the result of the count from the SELECT section
END;
